int main() { 
  

  FILE *f;
  char c[1000];
  f = fopen("test_file.txt", "r");
  while(!feof(f)) {
    if (fgets(c, 1000, f) != NULL) {
    // fscanf(f, "%s", c); читает одно слово. фгетс читает строку
    printf("%s", c);
    }
  }
  fclose(f);

  return 0;
